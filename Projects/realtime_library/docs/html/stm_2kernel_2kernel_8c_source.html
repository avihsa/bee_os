91void kernel_update(void) {
92
93 // update system components
94 HAL_IncTick();
95 // exit immediately, if kernel is not running yet
96 if (g_running_task_current == NULL
97 || g_kernel_status == EN_KERNEL_NOT_INITIALIZED
98 || g_kernel_status == EN_KERNEL_ERROR) {
99 return;
100 }
101
102 // inform segger systick interrupt aka kernel_update was entered
103 SEGGER_SYSVIEW_RECORD_ENTER_ISR();
104
105 // Return during critical section, but only after recording for SEGGER
106 if (g_kernel_critical_section_active) {
107 SEGGER_SYSVIEW_RECORD_EXIT_ISR();
108 return;
109 }
110
111 // handle delta times in delayed task list
112 size_t status = -1;
113 task_t *task = NULL;
114 if (g_delayed_tasks->size>0) {
115 task = (task_t *) g_delayed_tasks->tail->data;
116 if (task->delta_time==0) {
117 // reinsert task from delay list to its priority group
118 status = kernel_reinsert_task(&g_delayed_tasks, &g_delayed_tasks->tail, &task);
119 }
120 else {
121 // decrement delta time
122 task->delta_time--;
123 }
124 }
125
126
127 // check if time quantum has elapsed and update kernel state
128 if (g_running_task_current->time_quantum_remaining == 0
129 && !g_kernel_critical_section_active
130 && g_kernel_status != EN_KERNEL_IDLE) {
131 kernel_start_task(&g_priority_group_next, &g_linked_list_task_iterator_next, &g_running_task_next);
132 }
133 else if (g_kernel_status == EN_KERNEL_IDLE && status == KERNEL_SUCCESS) {
134 // exit idle
135 kernel_exit_idle();
136 }
137 else if (g_running_task_current->time_quantum_remaining > 0) {
138 // decrement time quantum
139 g_running_task_current->time_quantum_remaining--;
140 }
141
142 // inform segger systick interrupt aka kernel_update is about to exit
143 SEGGER_SYSVIEW_RECORD_EXIT_ISR();
144}

151__attribute__((naked))
152void kernel_schedule_task(void) {
153
154 // 1. PendSv pushes stack frame on stack
155
156 // 2. allocate stack memory for following operations and disable interrupts
157 __asm volatile ("PUSH {LR}");
158 // move stack pointer enough for variables
159 __asm volatile ("SUB SP, SP, #8");
160 // after moving the stackpointer functions and everything else can be done
161 kernel_disable_interrupts();
162 size_t status = 0;
163 uint32_t psp;
164
165 // 3. push R4-R11 on stack
166 __asm volatile ("PUSH {R2}");
167 __asm volatile ("MRS R2, PSP");
168 __asm volatile ("STMDB R2!, {R4-R11}");
169 __asm volatile ("MSR PSP, R2");
170
171 // update system
172 __ISB();
173
174 // obtain psp
175 psp = __get_PSP();
176
177
178 // 4. load next task by using its psp and move memory location
179
180 task_reset_time_quantum_remaining(&g_running_task_current);
181
182 status = task_checking(&g_running_task_previous);
183 if (status == TASK_SUCCESS) {
184 // save previous tasks psp and update segger stack usage
185 g_running_task_previous->task_data->u32TaskSP = psp;
186 SEGGER_SET_STACKPOINTER(g_running_task_previous);
187 }
188
189 // get tasks psp and prepare for reentry
190 psp = g_running_task_current->task_data->u32TaskSP;
191 task_set_state(&g_running_task_current, TaskState_Running);
192
193
194
195 __set_PSP(psp);
196
197 // check if critical section is still set
198 if (g_kernel_critical_section_active) {
199 kernel_toggle_critical_section();
200 }
201
202 // pop R4-R11 from Stack in reverse order
203
204 //__set_CONTROL(__get_CONTROL() | (1 << 1));
205 __asm volatile ("MRS R2, PSP");
206 __asm volatile ("LDMIA R2!, {R4-R11}");
207
208 __asm volatile ("MSR PSP, R2");
209 __asm volatile ("POP {R2}");
210 __ISB();
211
212
213 // 5. enable interrupts
214 kernel_enable_interrupts();
215 // free up stack and undo changes
216 __asm volatile ("ADD SP, SP, #8");
217 __asm volatile ("POP {LR}");
218
219 // 6. leave PendSv Routine
220 __asm volatile ("BX LR");
221}

231size_t kernel_swap_task(linked_list_t** priority_group, linked_list_element_t **linked_list_element, task_t **task) {
232
233 // set task to blocked and check for errors
234 task_t *task_blocked = (*task);
235 size_t status = task_set_state(task, TaskState_Blocked);
236 if (status != TASK_SUCCESS) {
237 kernel_set_status(EN_KERNEL_ERROR);
238 return ERROR_INFO(status, KERNEL_TASK_ERROR_REGISTER, KERNEL_UNABLE_TO_SCHEDULE_TASK);
239 }
240
241 // check if next task was set
242 if ((*priority_group)->size == 0
243 || g_linked_list_task_iterator == NULL
244 || g_running_task_next == NULL) {
245
246 // determine next task
247 linked_list_t *next_priority_group = NULL;
248 do {
249 status = dictionary_get(&g_prioritized_tasks, g_dictionary_priority_next, (void **) &next_priority_group);
250 g_dictionary_priority_next++;
251 } while (next_priority_group->size == 0 && g_dictionary_priority_next < KERNEL_MAX_TASK);
252
253 if (next_priority_group->size == 0 && g_dictionary_priority_next == KERNEL_MAX_TASK) {
254 // unable to find executable task in any priority, enter idle mode
255 kernel_enter_idle();
256 // assume next task was set by reinsert
257 }
258 else {
259 // Make sure we are in a critical section
260 if (!g_kernel_critical_section_active) {
261 kernel_toggle_critical_section();
262 }
263
264 // update next task information
265 g_dictionary_priority = g_dictionary_priority_next;
266 g_priority_group_next = next_priority_group;
267 g_linked_list_task_iterator_next = g_priority_group_next->tail;
268 g_running_task_next = (task_t *) g_linked_list_task_iterator_next->data;
269 }
270 }
271
272 // start next task and check for errors
273 status = kernel_start_task(&g_priority_group_next, &g_linked_list_task_iterator_next, &g_running_task_next);
274 if (status != KERNEL_SUCCESS) {
275 return ERROR_INFO(status, KERNEL_TASK_ERROR_REGISTER, KERNEL_UNABLE_TO_SWAP);
276 }
277
278 // catch blocked task
279 while(task_blocked->task_data->eTaskState == TaskState_Blocked);
280 return KERNEL_SUCCESS;
281}

319 // implements priority inheritance
320 /*
321 * Inheritance starts with lowest priority group
322 * and all tasks are moved higher at the same time.
323 * To prevent starvation for higher task a cool down is needed,
324 * because there can be more lower prioritized tasks.
325 */
326 static size_t inheritance_cooldown = 0;
327
328 // lower_priority might point to the last used priority group
329 // it needs to be prevented that the new priority group is being moved to a higher priority group by inheritance
330 if (g_task_lower_priority <= g_dictionary_priority) {
331 g_task_lower_priority = g_task_lowest_priority;
332 }
333
334 // move priority groups, if the current priority group is lower
335 if (inheritance_cooldown == 0 && g_task_lower_priority > g_dictionary_priority) {
336
337 // obtain priority group and check for errors
338 linked_list_t *lower_priority_group = NULL;
339 status = dictionary_get(&g_prioritized_tasks, g_task_lower_priority, (void **) &lower_priority_group);
340 if (status != DICTIONARY_SUCCESS) {
341 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_SCHEDULE_TASK);
342 }
343
344 // obtain higher priority group and check for errors
345 g_task_lower_priority--;
346 linked_list_t *higher_priority_group = NULL;
347 status = dictionary_get(&g_prioritized_tasks, g_task_lower_priority, (void **) &higher_priority_group);
348 if (status != DICTIONARY_SUCCESS) {
349 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_SCHEDULE_TASK);
350 }
351
352 // check if current priority group was reached and move priority group
353 size_t moved_elements = lower_priority_group->size;
354
355 linked_list_move_linked_list_after(&higher_priority_group, &lower_priority_group);
356
357 if (g_dictionary_priority == g_task_lower_priority) {
358 // reset inheritance and set cool down
359 g_task_lower_priority = g_task_lowest_priority;
360 inheritance_cooldown = moved_elements;
361 }
362 }
363
364 // set current task information and preset next task
365 g_running_task_current = *task;
366 g_linked_list_task_iterator = *linked_list_element;
367 g_priority_group_current = *priority_group;
368 g_priority_group_next = *priority_group;
369
370 if (g_linked_list_task_iterator != NULL) {
371 // set next task
372 g_linked_list_task_iterator_next = g_linked_list_task_iterator->next;
373 if (g_linked_list_task_iterator_next == NULL) {
374 // end of priority group and restart with first element
375 g_linked_list_task_iterator_next = g_priority_group_current->tail;
376 }
377 g_running_task_next = (task_t *) g_linked_list_task_iterator_next->data;
378 g_dictionary_priority_next = g_dictionary_priority + 1;
379 }
380 else if (g_priority_group_current->size == 0) {
381 // jump to next priority group
382 dictionary_get(&g_prioritized_tasks, g_dictionary_priority_next, (void **) &g_priority_group_current);
383 g_priority_group_next = g_priority_group_current;
384 g_linked_list_task_iterator = g_priority_group_current->tail;
385 g_running_task_current = (task_t *) g_linked_list_task_iterator;
386 g_dictionary_priority = g_dictionary_priority_next;
387 }
388 else {
389 // end of priority group and restart with first element
390 g_linked_list_task_iterator = g_priority_group_current->tail;
391 g_running_task_current = (task_t *) g_linked_list_task_iterator->data;
392 g_linked_list_task_iterator_next = g_linked_list_task_iterator->next;
393 g_running_task_next = (task_t *) g_linked_list_task_iterator_next->data;
394 g_dictionary_priority_next = g_dictionary_priority + 1;
395 }
396
397 // ------------------- critical section end ----------------------------
398 SCB->ICSR |= SCB_ICSR_PENDSVSET_Msk; // set PendSV-Flag
399
400 kernel_enable_interrupts();
401
402 return KERNEL_SUCCESS;
403}

410size_t kernel_set_status(Kernel_Status_e status) {
411
412 // finite state machine for kernel states
413 // it blocks on wrong transition or on an error
414 switch (status) {
415 case EN_KERNEL_STARTING:
416 if (g_kernel_status != EN_KERNEL_NOT_INITIALIZED) {
417 while (true);
418 }
419 break;
420 case EN_KERNEL_RUNNING:
421 if (!(g_kernel_status == EN_KERNEL_STARTING || g_kernel_status == EN_KERNEL_IDLE)) {
422 while (true);
423 }
424 break;
425 case EN_KERNEL_IDLE:
426 if (g_kernel_status != EN_KERNEL_RUNNING) {
427 while (true);
428 }
429 break;
430 default :
431 while(true);
432 }
433
434 g_kernel_status = status;
435
436 return KERNEL_SUCCESS;
437}
