47size_t queue_create(queue_t **queue, size_t queue_size, size_t element_size) {
48
49 // allocate memory and return on error
50 *queue = (queue_t *) malloc(sizeof(queue_t));
51 if ((*queue)==NULL) {
52 return QUEUE_NO_MEMORY;
53 }
54
55 // default initialization
56 (*queue)->size = queue_size;
57 (*queue)->length = 0;
58 (*queue)->head = (*queue)->size-1;
59 (*queue)->tail = 0;
60 (*queue)->element_size = element_size;
61
62 // create data entries
63 (*queue)->data = (void **) malloc((*queue)->size*sizeof(void *));
64 if ((*queue)->data==NULL) {
65 return QUEUE_DATA_NO_MEMORY;
66 }
67
68 // assign continuous memory space
69 (*queue)->data[0] = malloc((*queue)->size*(*queue)->element_size);
70 for (size_t i = 1; i<(*queue)->size; i++) {
71 // allow pointer arithmetic
72 (*queue)->data[i] = (char *)(*queue)->data[0] + i * (*queue)->element_size;
73
74 if ((*queue)->data[i]==NULL) {
75 return QUEUE_DATA_NO_MEMORY;
76 }
77 }
78
79 return QUEUE_SUCCESS;
80}

206size_t queue_read(queue_t **queue, void **element) {

208 // check queue for irregular structure

210 if (status!=QUEUE_SUCCESS) {
211 return status;
212 }
213
214 // it is not possible to read on empty queue
215 if ((*queue)->length==0) {
216 element = NULL;
217 return QUEUE_NO_ELEMENT;
218 }
219
220 // calculate next tail position and jump back if necessary
221 size_t next_position = ((*queue)->tail + 1) % (*queue)->size;
222 // copy data and check for error
223 void *destination = memcpy(*element, (*queue)->data[(*queue)->tail], (*queue)->element_size);
224 if (destination != (*element)) {
225 return QUEUE_COPY_ERROR;
226 }
227
228 // update queue
229 (*queue)->tail = next_position;
230 (*queue)->length--;
