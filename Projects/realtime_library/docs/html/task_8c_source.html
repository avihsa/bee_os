158size_t task_set_state(task_t **task, TCB_eTastStates_t task_state) {
159 size_t status = task_checking(task);
160 if (status!=TASK_SUCCESS) {
161 return status;
162 }
163
164 (*task)->task_data->eTaskState = task_state;
165
166 switch (task_state) {
167 case TaskState_Created:
168 SEGGER_SYSVIEW_TASK_CREATE((*task)->info.TaskID);
169 SEGGER_SYSVIEW_SEND_TASK_INFO(&((*task)->info));
170 break;
171 case TaskState_Ready:
172 SEGGER_SYSVIEW_TASK_START_READY((*task)->info.TaskID);
173 break;
174 case TaskState_Running:
175 SEGGER_SYSVIEW_TASK_START_EXEC((*task)->info.TaskID);
176 break;
177 case TaskState_Blocked:
178 SEGGER_SYSVIEW_TASK_STOP_READY((*task)->info.TaskID, (*task)->cause);
179 break;
180 case TaskState_Deleted:
181 SEGGER_SYSVIEW_TASK_STOP_EXEC();
182 break;
183 default:
184 return TASK_UNDEFINED_STATE;
185 }
186
187 return TASK_SUCCESS;
188}
