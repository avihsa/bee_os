95 if (status != LINKED_LIST_SUCCESS) {
96 return status;
97 }
98
99 // delete all linked list elements

101 linked_list_element_t *tail = (*linked_list)->tail;
102 (*linked_list)->tail = (*linked_list)->tail->next;
103 tail->previous = NULL;
104 free(tail);
105 tail = NULL;
106 (*linked_list)->size--;
107 }
108
109 // check for irregular structure
110 if ((*linked_list)->size != 0) {
111 return LINKED_LIST_IRREGULAR_STRUCTURE;
112 }

115 free((*linked_list));
116 (*linked_list) = NULL;

344size_t linked_list_pop_inner(linked_list_t **linked_list, linked_list_element_t **linked_list_element, void **data) {
345
346 // check linked list for irregular structure
347 size_t status = linked_list_checking(linked_list);
348 if (status != LINKED_LIST_SUCCESS) {
349 return status;
350 }
351
352 // obtain data
353 linked_list_element_t *pop = *linked_list_element;
354 if (pop == NULL) {
355 return LINKED_LIST_NO_ELEMENT;
356 }
357 (*data) = pop->data;
358
359 // remove data
360 if ((*linked_list)->size == 0 && (*linked_list)->tail == (*linked_list)->head) {
361 // linked list is empty
362 (*linked_list)->tail = NULL;
363 (*linked_list)->head = NULL;
364 }
365 else if ((*linked_list)->size == 0 && (*linked_list)->tail != (*linked_list)->head) {
366 // error
367 return LINKED_LIST_IRREGULAR_STRUCTURE;
368 }
369 else {
370 if (*linked_list_element == (*linked_list)->head) {
371 // simplify with pop front
372 return linked_list_pop_front(linked_list, data);
373 }
374 else if (*linked_list_element == (*linked_list)->tail) {
375 // simplify with pop back
376 return linked_list_pop_back(linked_list, data);
377 }
378 else {
379 // remove element in between
380 (*linked_list_element)->next->previous = (*linked_list_element)->previous;
381 (*linked_list_element)->previous->next = (*linked_list_element)->next;
382 }
383 }
384
385 // decrease size, when the element was remove in between and not by pop
386 (*linked_list)->size--;
387
388 // delete linked list element
389 free(pop);
390 pop = NULL;
391
392 return LINKED_LIST_SUCCESS;

468 }

470 // linked list element is between or tail
471 linked_list_element_tranfer->next->previous = linked_list_element_tranfer->previous;
472 }
473
474 if (linked_list_element_tranfer == (*linked_list_source)->tail) {
475 // linked list element is sources tail
476 (*linked_list_source)->tail = (*linked_list_source)->tail->next;
477 if ((*linked_list_source)->tail != NULL) {
478 // source is not empty
479 (*linked_list_source)->tail->previous = NULL;
480 }
481 }
482 else if (linked_list_element_tranfer->previous != NULL) {
483 // linked list element is between or head
484 linked_list_element_tranfer->previous->next = linked_list_element_tranfer->next;
485 }
486
487 // linked list element was removed
488 (*linked_list_source)->size--;
489
490 // set linked list element to default before insert in destination
491 linked_list_element_tranfer->next = NULL;
492 linked_list_element_tranfer->previous = NULL;
493
494 // check if linked list destination is empty
495 if ((*linked_list_destination)->size == 0) {
496 (*linked_list_destination)->tail = linked_list_element_tranfer;
497 (*linked_list_destination)->head = linked_list_element_tranfer;
498 }
499 else {
500 // linked list destination is not empty
501 (*linked_list_destination)->head->next = linked_list_element_tranfer;
502 linked_list_element_tranfer->previous = (*linked_list_destination)->head;
503 (*linked_list_destination)->head = linked_list_element_tranfer;
504 }
505
506 (*linked_list_destination)->size++;
507
508 return LINKED_LIST_SUCCESS;

624size_t linked_list_move_linked_list_after(
625 linked_list_t **linked_list_destination,
626 linked_list_t **linked_list_source) {
627
628 // check linked list for irregular structure
629 size_t status = linked_list_checking(linked_list_destination);
630 if (status != LINKED_LIST_SUCCESS) {
631 return status;
632 }
633
634 // check linked list for irregular structure
635 status = linked_list_checking(linked_list_source);
636 if (status != LINKED_LIST_SUCCESS) {
637 return status;
638 }
639
640 // abort transfer if empty
641 if ((*linked_list_source)->size == 0) {
642 return LINKED_LIST_SOURCE_IS_EMPTY;
643 }
644
645 // check where to move the source linked list to the destination linked list
646 if ((*linked_list_destination)->size == 0) {
647 // the destination linked list becomes the source linked list, without losing the source linked list content
648 (*linked_list_destination)->tail = (*linked_list_source)->tail;
649 }
650 else {
651 // the linked lists are being concatenated in linked list destination
652 (*linked_list_destination)->head->next = (*linked_list_source)->tail;
653 (*linked_list_source)->tail->previous = (*linked_list_destination)->head;
654 }
655
656 // update new head
657 (*linked_list_destination)->head = (*linked_list_source)->head;
658
659 // remove content from source and update each sizes
660 (*linked_list_source)->tail = NULL;
661 (*linked_list_source)->head = NULL;
662
663 (*linked_list_destination)->size += (*linked_list_source)->size;
664 (*linked_list_source)->size = 0;
665
666 return LINKED_LIST_SUCCESS;
667}
