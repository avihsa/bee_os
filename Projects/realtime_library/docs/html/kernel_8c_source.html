73int kernel_init(void) {
74 // set relevant system functions, depending on the used platform
75 kernel_set_system_functions();
76 // create prioritized task list with groups
77 int status = dictionary_create(&g_prioritized_tasks, KERNEL_MAX_TASK);

79 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_NO_PRIORITIZED_TASKS);
80 }
81
82 // add priority groups to priority list
83 linked_list_t *priority_group = NULL;
84 for (size_t priority = 0; priority<KERNEL_MAX_TASK; priority++) {
85
86 status = linked_list_create(&priority_group);
87 if (status!=LINKED_LIST_SUCCESS) {
88 return ERROR_INFO(status, KERNEL_LINK_LIST_ERROR_REGISTER, KERNEL_NO_PRIORITIZED_TASKS);

90
91 status = dictionary_add(&g_prioritized_tasks, priority, (void **) &priority_group);
92 if (status!=DICTIONARY_SUCCESS) {
93 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_NO_PRIORITIZED_TASKS);
94 }
95
96 }
97
98
99 // create linked list which lists all delayed tasks
100 status = linked_list_create(&g_delayed_tasks);
101 if (status!=LINKED_LIST_SUCCESS) {
102 return ERROR_INFO(status, KERNEL_LINK_LIST_ERROR_REGISTER, KERNEL_NO_DELAYED_TASKS);

104
105

107 status = dictionary_create(&g_message_queue_list, KERNEL_MAX_MESSAGE_QUEUE);

135int kernel_add_task(int (*task_main)(void), uint8_t u8_task_id, uint8_t u8_task_priority, size_t time_quantum) {
136
137 // create a mandatory task structure as base for the tcb and relevant kernel attributes
138 task_t *task = NULL;
139 int status = task_create(&task, task_main, u8_task_id, u8_task_priority, time_quantum);
140 if (status!=TASK_SUCCESS) {
141 return ERROR_INFO(status, KERNEL_TASK_ERROR_REGISTER, KERNEL_UNABLE_TO_ADD_TASK);
142 }
143
144 // insert the task in the corresponding priority group
145 linked_list_t *priority_group = NULL;
146 status = dictionary_get(&g_prioritized_tasks, u8_task_priority, (void **) &priority_group);
147 if (status!=DICTIONARY_SUCCESS) {
148 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_ADD_TASK);
149 }
150
151 // always push a task at the linked list head, because the kernel uses the tail by default as first
152 status = linked_list_push_front(&priority_group, (void **) &task);
153 if (status!=LINKED_LIST_SUCCESS) {
154 return ERROR_INFO(status, KERNEL_LINK_LIST_ERROR_REGISTER, KERNEL_UNABLE_TO_ADD_TASK);
155 }
156
157 return KERNEL_SUCCESS;
158}

168int kernel_start(void){
169 task_t *task = NULL;
170 linked_list_t *priority_group = NULL;
171 linked_list_element_t *task_iterator = NULL;
172 // set the alternative stack pointer to a default position
173 kernel_set_stack_pointer();
174 kernel_set_status(EN_KERNEL_STARTING);
175 // iterate over all priority lists and start all containing tasks
176 int status = 0;
177 for (size_t task_priority=0; task_priority<TASK_MAX_PRIOTITY; task_priority++) {
178
179 // check, if the priority group was actually set or abort on any other error
180 status = dictionary_get(&g_prioritized_tasks, task_priority, (void **) &priority_group);
181 if (status==DICTIONARY_VALUE_IS_NULL) {
182 continue;
183 }
184 else if (status!=DICTIONARY_SUCCESS) {
185 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_ADD_TASK);
186 }
187
188 // iterate over the priority group from the prioritized task list and start the task
189 task_iterator = priority_group->tail;
190 while (task_iterator!=NULL) {
191 task = (task_t *) task_iterator->data;
192
193 // check task before any use
194 status = task_checking(&task);
195 if (status==TASK_SUCCESS) {
196 status = kernel_start_task(&priority_group, &task_iterator, &task, TaskState_Ready);
197 if (status!=KERNEL_SUCCESS) {
198 return status;
199 }
200 }
201
202 task_iterator = task_iterator->next;
203 }
204 }
205
206 // it will return if the system is capable of running tasks in parallel
207 return KERNEL_SUCCESS;
208}

210/* @brief Create a new message queue for interprocess communication.
211 * @param message_queue_identifier is a pointer of pointer to the message queue identifier, which is being used as a key to the message queue
212 * @param name is a char pointer to a string, which can be used to identify the message queue
213 * @param queue_size is a size_t for the amount of entries in a message queue
214 * @param element_size is a size_t, which determines the size of each entry in the message queue
215 * @return 0 on success or greater 0 on error
216 * @info the return value is a concatenated status error code based of subcomponents.
217 **/
218int kernel_message_queue_create(message_queue_identifier_t **message_queue_identifier, char *name, size_t queue_size, size_t element_size) {
219 // return immediately if the amount of message queue exceeded
220 if (g_message_queue_ids>=KERNEL_MAX_MESSAGE_QUEUE) {
221 return KERNEL_UNABLE_TO_ADD_MESSAGE_QUEUE;
222 }
223
224 // create a new message on the provided parameters
225 message_queue_t *message_queue = NULL;
226 int status = message_queue_create(&message_queue, queue_size, element_size, g_message_queue_ids, name);
227 if (status!=MESSAGE_QUEUE_SUCCESS) {
228 return ERROR_INFO(status, KERNEL_MESSAGE_QUEUE_ERROR_REGISTER, KERNEL_UNABLE_TO_ADD_MESSAGE_QUEUE);
229 }
230
231 // add created message queue in a dictionary for fast access
232 status = dictionary_add(&g_message_queue_list, g_message_queue_ids, (void **) &message_queue);
233 if (status!=MESSAGE_QUEUE_SUCCESS) {
234 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_ADD_MESSAGE_QUEUE);
235 }
236
237 // increment the amount of message queue to limit the amount
238 g_message_queue_ids++;
239
240 // set the key to the message queue
241 (*message_queue_identifier) = message_queue->message_queue_identifier;
242
243 return KERNEL_SUCCESS;
244}

375int kernel_semaphore_acquire(size_t id) {
376
377 if (id >= KERNEL_MAX_SEMAPHORE) {
378 return KERNEL_UNABLE_TO_ACQUIRE_SEMAPHORE;
379 }
380
381 // prevent other tasks of manipulating the semaphore by blocking the context switch
382 // ------------------- critical section start -------------------------
383 kernel_toggle_critical_section();
384
385 // get the requested semaphore by the id from the semaphore dictionary
386 semaphore_t* semaphore = NULL;
387 int status = dictionary_get(&g_semaphore_list, id, (void **) &semaphore);
388 if (status != DICTIONARY_SUCCESS) {
389 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_ACQUIRE_SEMAPHORE);
390 }
391
392 // try to obtain a token from the semaphore
393 do {
394
395 status = semaphore_acquire(&semaphore, &g_priority_group_current, &g_linked_list_current_running_task, &g_running_task_current);
396
397 if (status == SEMAPHORE_SUCCESS) {
398 // acquired semaphore and semaphore will not be modified
399 kernel_toggle_critical_section();
400 // ------------------- critical section end ----------------------------
401 break;
402 }
403 else if (status == SEMAPHORE_NO_TOKENS) {
404 // task was inserted in the semaphore waiting list
405 kernel_swap_task(&g_priority_group_current, &g_linked_list_current_running_task, &g_running_task_current);
406 }
407 else {
408 // error happened with the semaphore
409 return ERROR_INFO(status, KERNEL_SEMAPHORE_ERROR_REGISTER, KERNEL_UNABLE_TO_ACQUIRE_SEMAPHORE);
410 }
411 } while (status == SEMAPHORE_NO_TOKENS);
412
413 return KERNEL_SUCCESS;
414}

423int kernel_semaphore_release(size_t id) {
424 if (id >= KERNEL_MAX_SEMAPHORE) {
425 return KERNEL_UNABLE_TO_ACQUIRE_SEMAPHORE;
426 }
427
428 // prevent other tasks of manipulating the semaphore by blocking the context switch
429 // ------------------- critical section start -------------------------
430 kernel_toggle_critical_section();
431
432 // get the requested semaphore by the id from the semaphore dictionary
433 semaphore_t* semaphore = NULL;
434 int status = dictionary_get(&g_semaphore_list, id, (void **) &semaphore);
435 if (status != DICTIONARY_SUCCESS){
436 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_ACQUIRE_SEMAPHORE);
437 }
438
439 // release the semaphore and check, if a task can be moved from the waiting list to the corresponding priority group
440 task_t* task = NULL;
441 linked_list_element_t* element = NULL;
442 status = semaphore_release(&semaphore, &element, &task);
443
444 if (task!=NULL) {
445
446 // get the priority group for the task
447 linked_list_t* priority_group = NULL;
448 status = dictionary_get(&g_prioritized_tasks, task->task_data->u8TaskPrio, (void **) &priority_group);
449 if (status != DICTIONARY_SUCCESS) {
450 return ERROR_INFO(status, KERNEL_DICTIONARY_ERROR_REGISTER, KERNEL_UNABLE_TO_RELEASE_SEMAPHORE);
451 }
452
453
454 // transfer the task from the semaphore waiting list to its priority group
455 status = linked_list_transfer(&priority_group, &(semaphore->task_waiting_list), &element);
456 if (status != LINKED_LIST_ELEMENT_SUCCESS) {
457 return ERROR_INFO(status, KERNEL_SEMAPHORE_ERROR_REGISTER, KERNEL_UNABLE_TO_RELEASE_SEMAPHORE);
458 }
459
460 // mark task as ready again
461 task_set_state(&task, TaskState_Ready);
462
463 // it is important to update the next task logic, if the moved task belongs to the current running priority group
464 g_linked_list_task_iterator_next = g_linked_list_task_iterator->next;
465 g_running_task_next = g_linked_list_task_iterator_next->data;
466
467
468 }
469 if (status != SEMAPHORE_SUCCESS) {
470 return ERROR_INFO(status, KERNEL_SEMAPHORE_ERROR_REGISTER, KERNEL_UNABLE_TO_ACQUIRE_SEMAPHORE);
471 }
472
473 kernel_toggle_critical_section();
474 // ------------------- critical section end ----------------------------
475
476
477 return KERNEL_SUCCESS;
