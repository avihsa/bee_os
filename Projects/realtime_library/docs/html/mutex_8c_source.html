106size_t mutex_acquire(mutex_t **mutex, linked_list_t **running_task_list, linked_list_element_t **running_task_element, task_t **task) {
107 size_t status = mutex_checking(mutex);
108 if (status != MUTEX_SUCCESS) {
109 return status;
110 }
111
112 // Mutex can only be acquired if either:
113 // - it is not currently acquired (owner == NULL)
114 // - the owner tries to acquire it consecutively (recursive locking) (task == owner)
115 if (*task == (*mutex)->owner) {
116 (*mutex)->lock_count++;
117 }
118 else {
119 // semaphore_acquire catches the owner handling by design:
120 // If owner is NULL, we can just normally acquire the semaphore.
121 // If calling task is not the owner, semaphore returns NO_TOKENS and exits.
122 status = semaphore_acquire(&((*mutex)->binary_semaphore), running_task_list, running_task_element, task);
123 if (status == SEMAPHORE_NO_TOKENS) {
124 return MUTEX_OWNED_BY_OTHER_TASK;
125 }
126 else if (status != SEMAPHORE_SUCCESS) {
127 return ERROR_INFO(status, MUTEX_SEMAPHORE_ERROR_REGISTER, MUTEX_UNABLE_TO_ACQUIRE);
128 }
129
130 (*mutex)->owner = *task;
131 (*mutex)->lock_count++;
132 }
133
134 return MUTEX_SUCCESS;
135}

146size_t mutex_release(mutex_t **mutex, linked_list_element_t **element, task_t **task) {
147 size_t status = mutex_checking(mutex);
148 if (status != MUTEX_SUCCESS) {
149 return status;
150 }
151
152 if (*task != (*mutex)->owner) {
153 return MUTEX_OWNED_BY_OTHER_TASK;
154 }
155
156 if ((*mutex)->lock_count == 1) {
157 status = semaphore_release(&((*mutex)->binary_semaphore), element, task);
158 if (status != SEMAPHORE_SUCCESS) {
159 return ERROR_INFO(status, MUTEX_SEMAPHORE_ERROR_REGISTER, MUTEX_UNABLE_TO_RELEASE);
160 }
161
162 (*mutex)->owner = NULL;
163 }
164
165 (*mutex)->lock_count--;
166
167 return MUTEX_SUCCESS;
168}

246size_t mutex_checking(mutex_t **mutex) {
247 if (*mutex == NULL) {
248 return MUTEX_NO_MEMORY;
249 }
250
251 size_t status = 0;
252 status = semaphore_checking(&((*mutex)->binary_semaphore));
253 if (status != SEMAPHORE_SUCCESS) {
254 return ERROR_INFO(status, MUTEX_SEMAPHORE_ERROR_REGISTER, MUTEX_NO_SEMAPHORE);
255 }
256
257 if (( (*mutex)->owner == NULL && (*mutex)->lock_count != 0 )
258 || ( (*mutex)->owner != NULL && (*mutex)->lock_count == 0) ) {
259 return MUTEX_IRREGULAR_STRUCTURE;
260 }
261
262 return MUTEX_SUCCESS;
263}
